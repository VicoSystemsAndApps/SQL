SELECT COUNT(1) AS [texts per date],
YEAR(wdh_created_date) AS [year],
MONTH(wdh_created_date) AS [month],
SUM(isnull(wdh_cost,0)) AS [cost]
FROM [Pro2SQL_WDH].[dbo].[wdh_sms_details]
GROUP BY YEAR(wdh_created_date),
MONTH(wdh_created_date)
ORDER BY YEAR(wdh_created_date), MONTH(wdh_created_date);

SELECT * 
  FROM [Pro2SQL_WDH].[dbo].[wdh_sms_details]
  WHERE wdh_cost > 0;
