    DECLARE	@Subject		NVARCHAR(1000)
	,		@OperatorEmail	NVARCHAR(256)
	,		@ProfileName	NVARCHAR(256);
 
	SELECT	@OperatorEmail = email_address
	FROM	msdb.dbo.sysoperators
	WHERE	[name] = 'DocumotiveAlert';
 
	SELECT	*
	FROM	msdb.dbo.sysoperators
	WHERE	[name] = 'DocumotiveAlert';
 
	SELECT	@ProfileName = name
	FROM	msdb.dbo.sysmail_profile
	WHERE	profile_id = 1;
 
 
	EXEC msdb.dbo.sp_send_dbmail		@profile_name	= @ProfileName
									,	@recipients		=  @OperatorEmail
									,	@subject		= 'Test Job'
									,	@body			= 'Rays test Job';
